read_csv <- function(input_file){
    if (file.size(input_file) > 0){
      df <- read.table(input_file)
    }
  else{
    df <- data.frame(matrix(ncol = 1, nrow = 0))
    x <- c("V1")
    colnames(df) <- x
  }
  return(df)
}
